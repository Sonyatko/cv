                    template 
                    class Matrix {
                    private:
                      int n;
                      int m;
                      T** a;
                    public:
                      Matrix(int n, int m) {
                        this->n = n;
                        this->m = m;
                        this->a = new T*[n];
                        for (int i = 0; i < n; i++) {
                          a[i] = new T[m];
                        }
                      }
                      Matrix(const Matrix& matrixToCopy) {
                        this->n = matrixToCopy.n;
                        this->m = matrixToCopy.m;
                        this->a = new T[matrixToCopy.n];
                        for (int i = 0; i < matrixToCopy.n; i++) {
                          this->a[i] = new T[matrixToCopy.m];
                        }
                      }
                      ~Matrix() {
                        delete[]a;
                      }
                      friend ostream& operator<<(ostream& output, const Matrix& obj) {
                        for (int i = 0; i < n; i++) {
                          for (int j = 0; j < m; j++) {
                            output << obj.a[i][j] << " ";
                          }
                          cout << endl;
                        }
                        output << endl;
                        return output;
                      }
                      friend istream& operator>>(istream& input, Matrix& obj) {
                        for (int i = 0; i < n; i++) {
                          for (int j = 0; j < m; j++) {
                            input >> obj.a[i][j];
                          }
                        }
                        return input;

                        T summ = 0;
                        int counter = 0;
                        for (int i = 0; i < n; i++) {
                          for (int j = 0; j < m; j++) {
                            summ += a[i][j];
                            counter++;
                          }
                        }
